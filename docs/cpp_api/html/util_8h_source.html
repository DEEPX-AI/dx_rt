88 static inline void get_start_time(struct timeval *s)
89 {
90 #ifdef __linux__
91  gettimeofday(s, NULL);
92 #elif _WIN32
93  LARGE_INTEGER counter, freq;
94  QueryPerformanceCounter(&counter);
95  QueryPerformanceFrequency(&freq);
96  s->tv_sec = counter.QuadPart / freq.QuadPart;
97  s->tv_usec = (long)((counter.QuadPart % freq.QuadPart) * 1000000 / freq.QuadPart);
98 #endif
99 }
100 static inline uint64_t get_elapsed_time(struct timeval s)
101 {
102  struct timeval e;
103  uint64_t elapsed;
104 
105  get_start_time(&e);
106 
107  e.tv_sec = e.tv_sec - s.tv_sec;
108  if (e.tv_usec < s.tv_usec) {
109  e.tv_sec--;
110  e.tv_usec = (1000000 - s.tv_usec) + e.tv_usec;
111  } else {
112  e.tv_usec = e.tv_usec - s.tv_usec;
113  }
114 
115  elapsed = (e.tv_sec * 1000000) + e.tv_usec;
116 
117  return elapsed;
118 }
